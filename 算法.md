### 打印n*n矩阵的螺旋方形和回字形
```
private static void handle(String data) {
		int n = Integer.parseInt(data);
		int m = 0;
		int k = 0;
		// 存放螺旋方形
		int[][] a = new int[n][n];
		// 存放回字形
		int[][] b = new int[n][n];
		if (n % 2 == 0) {
			m = n;
		} else {
			m = n / 2 + 1;
		}

		for (int i = 0; i < m; i++) {
			for (int j = i; j < n - i; j++) {
				k++;
				a[i][j] = k;
			}
			for (int j = i + 1; j < n - i; j++) {
				k++;
				a[j][n - 1 - i] = k;
			}
			for (int j = n - i - 2; j >= i; j--) {
				k++;
				a[n - i - 1][j] = k;
			}
			for (int j = n - i - 2; j >= i + 1; j--) {
				k++;
				a[j][i] = k;
			}
		}
		System.out.println("打印螺旋方形");
		for (int i = 0; i < a.length; i++) {
			for (int j = 0; j < a.length; j++) {
				if (a[i][j] < 10) {
					System.out.print(" ");
				}

				System.out.print(a[i][j] + " ");
			}
			System.out.println();
		}

		System.out.println("---------");

		for (int i = 0; i < m; i++) {
			for (int j = i; j < n - i; j++) {
				b[i][j] = n - i;
			}
			for (int j = i + 1; j < n - i; j++) {
				b[j][n - i - 1] = n - i;
			}
			for (int j = n - i - 2; j >= i; j--) {
				b[n - i - 1][j] = n - i;
			}
			for (int j = n - i - 2; j >= i + 1; j--) {
				b[j][i] = n - i;
			}
		}
		System.out.println("打印回字形");

		for (int i = 0; i < a.length; i++) {
			for (int j = 0; j < a.length; j++) {

				System.out.print(b[i][j] + " ");
			}
			System.out.println();
		}

	}
```

```
	private static void merge(int[] data1, int[] data2) {

		int bLen = data2.length;
		int c = aLen + bLen;
		int[] data3 = new int[c];

		int i = 0, j = 0, k = 0;
```

### 两个数字字符串相加生成新的字符串 "11" "123" 相加生成"134"
```
private static void add(String data1, String data2) {
		// 依最大长度数字字符串长度为准
		int len = data1.length() > data2.length() ? data1.length() : data2.length();
		// 将长度短的数字字符串前面加0
		while (data1.length() != len) {
			data1 = 0 + data1;
		}
		while (data2.length() != len) {
			data2 = 0 + data2;
		}
		char[] d1 = data1.toCharArray();
		char[] d2 = data2.toCharArray();
		StringBuilder result = new StringBuilder();
		for (int i = 0; i < len; i++) {
			// 将char 转化为对应的int
			int x = d1[i] - '0';
			int y = d2[i] - '0';
			result.append((x + y));
		}
		System.out.println(result.toString());
	}
```
