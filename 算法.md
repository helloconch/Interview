### 打印n*n矩阵的螺旋方形和回字形
```
private static void handle(String data) {
		int n = Integer.parseInt(data);
		int m = 0;
		int k = 0;
		// 存放螺旋方形
		int[][] a = new int[n][n];
		// 存放回字形
		int[][] b = new int[n][n];
		if (n % 2 == 0) {
			m = n;
		} else {
			m = n / 2 + 1;
		}

		for (int i = 0; i < m; i++) {
			for (int j = i; j < n - i; j++) {
				k++;
				a[i][j] = k;
			}
			for (int j = i + 1; j < n - i; j++) {
				k++;
				a[j][n - 1 - i] = k;
			}
			for (int j = n - i - 2; j >= i; j--) {
				k++;
				a[n - i - 1][j] = k;
			}
			for (int j = n - i - 2; j >= i + 1; j--) {
				k++;
				a[j][i] = k;
			}
		}
		System.out.println("打印螺旋方形");
		for (int i = 0; i < a.length; i++) {
			for (int j = 0; j < a.length; j++) {
				if (a[i][j] < 10) {
					System.out.print(" ");
				}

				System.out.print(a[i][j] + " ");
			}
			System.out.println();
		}

		System.out.println("---------");

		for (int i = 0; i < m; i++) {
			for (int j = i; j < n - i; j++) {
				b[i][j] = n - i;
			}
			for (int j = i + 1; j < n - i; j++) {
				b[j][n - i - 1] = n - i;
			}
			for (int j = n - i - 2; j >= i; j--) {
				b[n - i - 1][j] = n - i;
			}
			for (int j = n - i - 2; j >= i + 1; j--) {
				b[j][i] = n - i;
			}
		}
		System.out.println("打印回字形");

		for (int i = 0; i < a.length; i++) {
			for (int j = 0; j < a.length; j++) {

				System.out.print(b[i][j] + " ");
			}
			System.out.println();
		}

	}
```
